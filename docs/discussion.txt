🗺️ 4. Kaartlagen integreren in Leaflet
js


Copy
function loadMapLayers(layers) {
  layers.forEach(layer => {
    if (layer.type === "point") {
      L.marker(layer.coords).addTo(map);
    }
    if (layer.type === "polyline") {
      L.polyline(layer.coords).addTo(map);
    }
    if (layer.type === "polygon") {
      L.polygon(layer.coords).addTo(map);
    }
  });
}

  const items = events.map(e => ({
    id: e.id,
    content: e.title,
    start: e.start,
    end: e.end
  }));
  new vis.Timeline(document.getElementById("timeline"), items);
}
